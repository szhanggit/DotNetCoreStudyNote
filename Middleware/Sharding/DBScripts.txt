CREATE   PROCEDURE [dbo].[sp_ins_user_with_roles] 
	-- Add the parameters for the stored procedure here
	@NewUsers dbo.tvp_ins_users READONLY,
	@UserRoles dbo.tvp_ins_user_roles_by_username READONLY,
	@UserCount INT = 0 OUTPUT,
	@UserRoleCount INT = 0 OUTPUT
AS
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	SET NOCOUNT, XACT_ABORT ON;

    BEGIN TRY

        BEGIN TRANSACTION;
		
			-- Insert into User Table first, DISTINCT users if not exist
			INSERT INTO dbo.tb_ubi_user_basic_information (user_name, email, firstname, lastname, is_active, creation_time)
			SELECT  DISTINCT UserName, Email, FirstName, LastName, 1, GETDATE()
			FROM    @NewUsers AS NewUsers
			WHERE   NOT EXISTS ( SELECT 1
								 FROM   dbo.tb_ubi_user_basic_information AS UBI
								 WHERE  UBI.user_name = NewUsers.UserName );

			SET @UserCount = @@ROWCOUNT
			
			-- Then, insert into User Role tables, by matching user id from users table
			INSERT INTO dbo.tb_ur_user_role (user_id, role_id, tenant_id)
			SELECT NewUsers.user_basic_info_id, TAU.RoleId, TAU.TenantId
			FROM (SELECT user_basic_info_id, user_name FROM dbo.tb_ubi_user_basic_information WHERE user_name IN (SELECT DISTINCT UserName FROM @NewUsers)) AS NewUsers
			RIGHT JOIN @UserRoles AS TAU
			ON NewUsers.user_name = TAU.UserName
			WHERE NOT EXISTS (SELECT * FROM dbo.tb_ur_user_role AS UR WHERE UR.user_id = NewUsers.user_basic_info_id AND UR.role_id = TAU.RoleId AND UR.tenant_id = TAU.TenantId)
			
			SET @UserRoleCount = @@ROWCOUNT
			
        COMMIT TRANSACTION;
		
    END TRY
    BEGIN CATCH 

       IF @@TRANCOUNT > 0
       BEGIN
          ROLLBACK TRANSACTION
       END;

		THROW
    END CATCH;

    SET NOCOUNT, XACT_ABORT OFF;
END

-- =============================================
-- Author:		Lean Alcabasa
-- Create date: 2021-09-20
-- Description:	This stored procedure will get users, and their roles per tenant
-- =============================================
CREATE PROCEDURE [dbo].[sp_sel_user_acess_list]
	@RowCount int = 20,
	@PageOffset int = 0,
	@TenantId int = NULL,
	@TotalCount int = 0 OUTPUT
AS
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	SET NOCOUNT ON;

	-- Return rows
	SELECT 
		UBI.user_basic_info_id "UserId", 
		UBI.user_name "UserName", 
		UBI.firstname "FirstName", 
		UBI.lastname "LastName", 
		UBI.email "Email", 
		Roles.role_id "RoleId", 
		Roles.displayname "RoleDisplayName", 
		Tenants.tenant_basic_info_id "TenantId", 
		Tenants.name  "TenantName",
		UBI.creation_time "CreationTime"
	FROM (SELECT 
		UBI.user_basic_info_id, 
		UR.role_id
	FROM tb_ur_user_role AS UR
	LEFT JOIN dbo.tb_ubi_user_basic_information UBI ON UR.user_id = UBI.user_basic_info_id
	WHERE (@TenantId IS NULL OR (UR.tenant_id = @TenantId)) AND UR.role_id > 1 -- Do not return Super Admins
	GROUP BY UBI.user_basic_info_id, UR.role_id
	ORDER BY UBI.user_basic_info_id DESC OFFSET @PageOffset ROWS
	FETCH NEXT @RowCount ROWS ONLY) AS UserRoleTenant
	LEFT JOIN dbo.tb_ur_user_role UR ON UserRoleTenant.user_basic_info_id = UR.user_id AND UserRoleTenant.role_id = UR.role_id
	LEFT JOIN dbo.tb_ubi_user_basic_information UBI ON UR.user_id = UBI.user_basic_info_id
	LEFT JOIN dbo.tb_tbi_tenant_basic_information Tenants ON UR.tenant_id = Tenants.tenant_basic_info_id
	LEFT JOIN dbo.tb_r_role Roles ON UR.role_id = Roles.role_id
	FOR JSON AUTO
	
	-- Count Total Rows
	SELECT 
		UBI.user_basic_info_id, 
		UR.role_id
	FROM tb_ur_user_role AS UR
	LEFT JOIN dbo.tb_ubi_user_basic_information UBI ON UR.user_id = UBI.user_basic_info_id
	WHERE (@TenantId IS NULL OR (UR.tenant_id = @TenantId))
	GROUP BY UBI.user_basic_info_id, UR.role_id
	ORDER BY UBI.user_basic_info_id DESC
	

	SET @TotalCount = @@ROWCOUNT
END

CREATE   PROCEDURE [dbo].[sp_ins_user_with_roles] 
	-- Add the parameters for the stored procedure here
	@NewUsers dbo.tvp_ins_users READONLY,
	@UserRoles dbo.tvp_ins_user_roles_by_username READONLY,
	@UserCount INT = 0 OUTPUT,
	@UserRoleCount INT = 0 OUTPUT
AS
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	SET NOCOUNT, XACT_ABORT ON;

    BEGIN TRY

        BEGIN TRANSACTION;
		
			-- Insert into User Table first, DISTINCT users if not exist
			INSERT INTO dbo.tb_ubi_user_basic_information (user_name, email, firstname, lastname, is_active, creation_time, check_sum)
			SELECT  DISTINCT UserName, Email, FirstName, LastName, 1, GETDATE(),null
			FROM    @NewUsers AS NewUsers
			WHERE   NOT EXISTS ( SELECT 1
								 FROM   dbo.tb_ubi_user_basic_information AS UBI
								 WHERE  UBI.user_name = NewUsers.UserName );

			SET @UserCount = @@ROWCOUNT
			
			-- Then, insert into User Role tables, by matching user id from users table
			INSERT INTO dbo.tb_ur_user_role (user_id, role_id, tenant_id)
			SELECT NewUsers.user_basic_info_id, TAU.RoleId, TAU.TenantId
			FROM (SELECT user_basic_info_id, user_name FROM dbo.tb_ubi_user_basic_information WHERE user_name IN (SELECT DISTINCT UserName FROM @NewUsers)) AS NewUsers

		_users.email,
		_users.creation_time

    SET @TotalCount = @@ROWCOUNT

-- =============================================
-- Author:		Lean Alcabasa
-- Create date: 2021-10-08
-- Description:	This stored procedure will get users, and their roles per tenant (excluding Super Admin roles)
-- =============================================
CREATE PROCEDURE [dbo].[sp_sel_user_acess_list]
	@RowCount int = 20,
	@PageOffset int = 0,
	@TenantId int = NULL,
	@TotalCount int = 0 OUTPUT
AS
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	SET NOCOUNT ON;

	-- Return rows
	SELECT 
		UBI.user_basic_info_id "UserId", 
		UBI.user_name "UserName", 
		UBI.firstname "FirstName", 
		UBI.lastname "LastName", 
		UBI.email "Email", 
		Roles.role_id "RoleId", 
		Roles.displayname "RoleDisplayName", 
		Tenants.tenant_basic_info_id "TenantId", 
		Tenants.name  "TenantName",
		UBI.creation_time "CreationTime"
	FROM (SELECT 
		UBI.user_basic_info_id, 
		UR.role_id
	FROM tb_ur_user_role AS UR
	LEFT JOIN dbo.tb_ubi_user_basic_information UBI ON UR.user_id = UBI.user_basic_info_id
	WHERE (@TenantId IS NULL OR (UR.tenant_id = @TenantId)) AND UR.role_id > 1 -- Do not return Super Admins
	GROUP BY UBI.user_basic_info_id, UR.role_id
	ORDER BY UBI.user_basic_info_id DESC OFFSET @PageOffset ROWS
	FETCH NEXT @RowCount ROWS ONLY) AS UserRoleTenant
	LEFT JOIN dbo.tb_ur_user_role UR ON UserRoleTenant.user_basic_info_id = UR.user_id AND UserRoleTenant.role_id = UR.role_id
	LEFT JOIN dbo.tb_ubi_user_basic_information UBI ON UR.user_id = UBI.user_basic_info_id
	LEFT JOIN dbo.tb_tbi_tenant_basic_information Tenants ON UR.tenant_id = Tenants.tenant_basic_info_id
	LEFT JOIN dbo.tb_r_role Roles ON UR.role_id = Roles.role_id
	FOR JSON AUTO
	
	-- Count Total Rows
	SELECT 
		UBI.user_basic_info_id, 
		UR.role_id
	FROM tb_ur_user_role AS UR
	LEFT JOIN dbo.tb_ubi_user_basic_information UBI ON UR.user_id = UBI.user_basic_info_id
	WHERE (@TenantId IS NULL OR (UR.tenant_id = @TenantId))
	GROUP BY UBI.user_basic_info_id, UR.role_id
	ORDER BY UBI.user_basic_info_id DESC
	

	SET @TotalCount = @@ROWCOUNT
END

-- Create date: 10/14/2021
-- Description:	This stored procedure was made to update the user tenants

-- Sample:
/*

EXEC [dbo].[sp_upd_user_tenants] @UserBasicInfoId =1
	, @TenantsPerRole =N'[{
    "roleId": 1,
    "tenants":[1,2]
},
{
    "roleId": 2,
    "tenants":[1,2,3]
}]'
*/
-- =============================================
CREATE PROCEDURE [dbo].[sp_upd_user_tenants]
	@UserBasicInfoId int,
	@TenantsPerRole nvarchar(max)
AS
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	SET NOCOUNT ON;


	DELETE FROM dbo.tb_ur_user_role where [user_id] = @UserBasicInfoId;

	INSERT INTO dbo.tb_ur_user_role([user_id], [role_id], tenant_id)
	SELECT
		@UserBasicInfoId
		, a.RoleId
		,Tenants = c.Value
	FROM
		OPENJSON(@TenantsPerRole)
		WITH
			(
				RoleId INT N'$.RoleId'
				,Tenants NVARCHAR(MAX) AS JSON
			) AS a
		CROSS APPLY
			OPENJSON(a.tenants) as c;

	SELECT @UserBasicInfoId;
END
GO

-- Author:		Lean Alcabasa
-- Create date: 10/12/2021
-- Description:	Select user by userId with roles. Added TenantId column.
-- =============================================
DROP PROCEDURE IF EXISTS [dbo].[sp_sel_user_acess_list]
GO
CREATE PROCEDURE [dbo].[sp_sel_user_acess_list]
	@RowCount int = 20,
	@PageOffset int = 0,
	@TenantId int = NULL,
	@TotalCount int = 0 OUTPUT,
	@Search nvarchar(128) = null
AS
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	SET NOCOUNT ON;

	-- Return rows
	SELECT 
		UBI.user_basic_info_id "UserId", 
		UBI.user_name "UserName", 
		UBI.firstname "FirstName", 
		UBI.lastname "LastName", 
		UBI.email "Email", 
		Roles.role_id "RoleId", 
		Roles.displayname "RoleDisplayName", 
		Tenants.tenant_basic_info_id "TenantId", 
		Tenants.name  "TenantName",
		UBI.creation_time "CreationTime"
	FROM (SELECT 
		UBI.user_basic_info_id, 
		UR.role_id
	FROM tb_ur_user_role AS UR
	LEFT JOIN dbo.tb_ubi_user_basic_information UBI ON UR.user_id = UBI.user_basic_info_id
	WHERE ((@TenantId IS NULL OR (UR.tenant_id = @TenantId)) AND UR.role_id > 1) -- Do not return Super Admins
	OR (
		(UBI.[user_name] like @Search + '%' OR @Search is null) OR
		(UBI.email like @Search + '%' OR @Search is null)
	)
	GROUP BY UBI.user_basic_info_id, UR.role_id
	ORDER BY UBI.user_basic_info_id DESC OFFSET @PageOffset ROWS
	FETCH NEXT @RowCount ROWS ONLY) AS UserRoleTenant
	LEFT JOIN dbo.tb_ur_user_role UR ON UserRoleTenant.user_basic_info_id = UR.user_id AND UserRoleTenant.role_id = UR.role_id AND (@TenantId IS NULL OR (UR.tenant_id = @TenantId))
	LEFT JOIN dbo.tb_ubi_user_basic_information UBI ON UR.user_id = UBI.user_basic_info_id
	LEFT JOIN dbo.tb_tbi_tenant_basic_information Tenants ON UR.tenant_id = Tenants.tenant_basic_info_id
	LEFT JOIN dbo.tb_r_role Roles ON UR.role_id = Roles.role_id
	FOR JSON AUTO
	
	-- Count Total Rows
	SELECT 
		UBI.user_basic_info_id, 
		UR.role_id
	FROM tb_ur_user_role AS UR
	LEFT JOIN dbo.tb_ubi_user_basic_information UBI ON UR.user_id = UBI.user_basic_info_id
	WHERE (@TenantId IS NULL OR (UR.tenant_id = @TenantId))
	GROUP BY UBI.user_basic_info_id, UR.role_id
	ORDER BY UBI.user_basic_info_id DESC
	

	SET @TotalCount = @@ROWCOUNT
END
GO

-- =============================================

CREATE PROCEDURE [dbo].[sp_create_program]
	@NewBarcodeSettings [dbo].[tvp_create_programbarcodesettings] READONLY
	,@Name					  	nvarchar(50)
	,@DisplayName			  	nvarchar(200)
	,@TimeOffset				varchar(6)
	,@MerchantMaxFailNumber		int
	,@ShopMaxFailNumber			int
	,@TerminalMaxFailNumber		int
	,@IsDefaultProgram			bit
	,@DateFormatString			varchar(25) = NULL
	,@DateTimeFormatString		varchar(25) = NULL
	,@IsEdenred					bit
	,@TenantId					int
	,@ProgramCollectionCode		nvarchar(255) = NULL
	,@IdentityCodeFormatted 	nvarchar(20)	= NULL OUTPUT
AS
BEGIN
    SET NOCOUNT, XACT_ABORT OFF;

	BEGIN TRANSACTION
    BEGIN TRY

	DECLARE @ProgramId	int
			, @IdentityCode int
			, @IdentityCodeStr nvarchar(255)
			, @AfterRemoveFistCharacter	nvarchar(255)
	
	SET @IdentityCodeStr = (SELECT TOP (1) [identity_code] FROM [dbo].[tb_p_program]
                            order by id desc)

	SET @AfterRemoveFistCharacter = (Select right(@IdentityCodeStr, len(@IdentityCodeStr)-1))  

	SET @IdentityCode = CAST(@AfterRemoveFistCharacter AS int) + 1;
	
	IF(@IdentityCode IS NULL)
		SET @IdentityCode = 1
		
	SET @IdentityCodeFormatted = CONCAT(@TenantId, FORMAT(@IdentityCode, '00000'))

	-- Insert into Program
		INSERT INTO [dbo].[tb_p_program]
				   ([identity_code]
				   ,[name]
				   ,[display_name]
				   ,[timeoffset]
				   ,[merchant_max_fail_number]
				   ,[shop_max_fail_number]
				   ,[terminal_max_fail_number]
				   ,[is_default_program]
				   ,[date_format_string]
				   ,[datetime_format_string]
				   ,[is_edenred]
				   ,[tenant_id])
			 VALUES
				   (@IdentityCodeFormatted
				   ,@Name
				   ,@DisplayName

	-- Add the parameters for the stored procedure here
	@tenant_basic_info_id INT OUTPUT,
	@name VARCHAR(100),
	@country_code VARCHAR(5),
	@timezone VARCHAR(100),
	@time_format VARCHAR(20),
	@currency_symbol VARCHAR(5),
	@company_tax_type BIT,
	@logo VARCHAR(MAX),
	@preferred_language VARCHAR(100),
	@company_tax_rate DECIMAL(18,2),
	@effectivity_date DATETIME
AS
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	SET NOCOUNT ON;

		SET NOCOUNT, XACT_ABORT ON;

    BEGIN TRY

        BEGIN TRANSACTION;
        

		    -- Insert statements for procedure here
    UPDATE dbo.tb_tbi_tenant_basic_information
	SET
	[name] = @name,
	country_code = @country_code,
	timezone = @timezone,
	time_format = @time_format,
	currency_symbol = @currency_symbol,
	company_tax_type = @company_tax_type,
	preferred_language = @preferred_language,
	logo = @logo
	WHERE tenant_basic_info_id = @tenant_basic_info_id

	--check if the tax rate is changed with effectivity date

	DECLARE @current_effectivity_date datetime
	DECLARE @current_tax_rate decimal(18,2)

	SELECT TOP 1 @current_tax_rate = company_tax_rate,
				 @current_effectivity_date = effectivity_date 
	FROM tb_tti_tenant_taxrate_information
    WHERE tenant_basic_info_id = @tenant_basic_info_id AND is_active = 1

	IF (@current_tax_rate <> @company_tax_rate) OR 
	   (@current_effectivity_date <> @effectivity_date) OR
	   (@current_effectivity_date IS NULL) OR (@current_tax_rate IS NULL)
		BEGIN

			UPDATE dbo.tb_tti_tenant_taxrate_information 
			SET 
			is_active = 0
			WHERE tenant_basic_info_id = @tenant_basic_info_id

			INSERT INTO dbo.tb_tti_tenant_taxrate_information
			(
				tenant_basic_info_id,
				effectivity_date,
				company_tax_rate,
				is_active
			)
			VALUES
			(
				@tenant_basic_info_id,
				@effectivity_date,
				@company_tax_rate,
				1
			)			
		END

        COMMIT TRANSACTION;

    END TRY
    BEGIN CATCH 

       IF @@TRANCOUNT > 0
       BEGIN
          ROLLBACK TRANSACTION
       END;

	   THROW;
    END CATCH;

    SET NOCOUNT, XACT_ABORT OFF;



END
